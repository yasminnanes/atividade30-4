Comandos do Github
  1. cd 'nome da pasta'
  2. git config --global user.name "seu nome"
  3. git config --global user.email "seu e-mail"
  4. git init
  5. git status
  6. git add .
  7. git commit _m "atualização"
  8. git remote add origin http://
  9. git push -u origin master
  10. git branch bug-123
  11. git checkout -b bug-456
  12. git pull
